## 1. SET UP POLARIS ENVIRONMENT 

cd $HOME
rm pixi
rm Miniforge3

showquota
# (make sure there is enough room for the install of polaris)
# if there isn't enough in $HOME:
    cd $SCRATCH

git clone git@github.com:E3SM-Project/polaris.git
cd polaris

# to avoid needing to enter your password multiple times:
    git config --global credential.helpter store
    eval "$(ssh-agent -s)"
    ssh-add ~/.ssh/id_ed25519 
    # or whatever ssh id you want to add
    # then enter your password when prompted

git submodule update --init --recursive

# perlmutter is a supported machine, so run the deploy command
./deploy.py --machine pm-cpu --compiler gnu --mpi mpich

# activate the environment
source $SCRATCH/polaris/load_polaris_pm-cpu_gnu_mpich.sh

## 2. SET UP BICHAN BRANCH

cd $HOME/repos/bichan_WMT_DVD

git clone -b bichan_10022025 git@github.com:dylanschlichting/E3SM.git

cd E3SM
# to avoid needing to enter your password multiple times:
    git config --global credential.helpter store
    eval "$(ssh-agent -s)"
    ssh-add ~/.ssh/id_ed25519 
    # or whatever ssh id you want to add
    # then enter your password when prompted

git submodule update --init --recursive

# activate the polaris environment 
source $SCRATCH/polaris/load_polaris_pm-cpu_gnu_mpich.sh

## 3. COMPILE MODEL
