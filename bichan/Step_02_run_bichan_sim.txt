## 2. CONFIGURE NAMELIST AND STREAMS FILE

## a) CONFIGURE NAMELIST
# recall: for these experiments, we want to turn on DVD and WMT diagnostics
# recall: for bichan specifically, we need to make sure that the harmonic and biharmonic viscosity is 
# twice that of what is prescribed by Kyle Hinson's paper
# Towards high-fidelity simulations of coastal submesoscale baroclinic instabilities with MPAS-O (vE3SM3.0.0) 
# Part I: Idealized experiments

# turn on DVD
&tracer_forcing_numericalMixingTracers
    config_use_numericalMixingTracers = .true.

&AM_discreteVarianceDecay
    config_AM_discreteVarianceDecay_enable = .true.
    config_AM_discreteVarianceDecay_compute_interval = 'dt'
    config_AM_discreteVarianceDecay_output_stream = 'discreteVarianceDecayOutput'
    config_AM_discreteVarianceDecay_compute_on_startup = .true.
    config_AM_discreteVarianceDecay_write_on_startup = .false.


# change the namelist to match the calculated parameters from the chan_args_namelists jupyter notebook
# note that we do not change config_tracer_del* because this changes kappa_h, which is not prescribed in bichan.

config_dt= 00: 0 : 24.0 
config_use_mom_del2 = .true.
config_mom_del2= 0.222 
config_use_mom_del4 = .true.
config_mom_del4= 3.1573e+05 
config_btr_dt= 0 : 2


# also check the stride and configuration in namelist.ocean
&io
    config_write_output_on_startup = .true.
    config_pio_num_iotasks = <nodes requested> # usually 1 or 4...
    config_pio_stride = 1 
/

# change the namelist "input files"

config_proc_decomp_file_prefix = '400m_channel_graph.info.part.'
